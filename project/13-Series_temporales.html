def verificar_biblioteca(biblioteca):
    try:

def instalar_una_biblioteca(biblioteca):
    resultado = verificar_biblioteca(biblioteca)
    if resultado[2]:
        display(HTML(f"<div class='info-box'> {biblioteca} ya est instalada - Versin: {resultado[1]}</div>"))
        return (biblioteca, True)
    try:
        subprocess.check_call([sys.executable, "-m", "pip", "install", biblioteca])
        display(HTML(f"<div class='info-box'> {biblioteca} instalada correctamente.</div>"))
        return (biblioteca, True)
    except Exception as e:
        manejar_error_importacion(biblioteca, str(e))
        return (biblioteca, False)

# Separar caractersticas y variable objetivo
X = df.drop('num_orders', axis=1)
y = df['num_orders']

# Identificar variables categricas y numricas
# En este dataset, despus del remuestreo, probablemente solo tengamos variables numricas
categorical_features = X.select_dtypes(include=['object']).columns.tolist()
numerical_features = X.select_dtypes(include=[np.number]).columns.tolist()

# Definir preprocesamiento para variables categricas y numricas
preprocessor = ColumnTransformer(

        ('num', StandardScaler(), numerical_features),
        # ('cat', OneHotEncoder(handle_unknown='ignore', sparse=False), categorical_features)  # Descomentar si hay variables categricas
    ]
)

#  4.3 Divisin del Conjunto de Datos

)
mostrar_mensaje(f" Datos divididos correctamente. Tamao del conjunto de entrenamiento: {X_train.shape[0]} registros.", "success")

#  4.4 Entrenamiento de Modelos
modelos = {
    'Regresin Lineal': LinearRegression(),
    'Bosque Aleatorio': RandomForestRegressor(random_state=42),
    'LightGBM': lgb.LGBMRegressor(random_state=42)
}

resultados = []

for nombre_modelo, modelo in modelos.items():
    mostrar_mensaje(f" Entrenando {nombre_modelo}...", "info")
    start_time = time.time()

    # Crear pipeline
    pipeline = Pipeline(steps=[
        ('preprocessor', preprocessor),
        ('regressor', modelo)
    ])

    # Ajuste de hiperparmetros
    if nombre_modelo == 'Bosque Aleatorio':
        param_grid = {
            'regressor__n_estimators': [100, 200],
            'regressor__max_depth': [10, 20, None],
            'regressor__min_samples_split': [2, 5]
        }
        grid_search = GridSearchCV(
            estimator=pipeline,
            param_grid=param_grid,
            cv=3,
            scoring='neg_mean_absolute_error',
            n_jobs=-1
        )
        grid_search.fit(X_train, y_train)
        best_model = grid_search.best_estimator_
        training_time = time.time() - start_time
        best_params = grid_search.best_params_
    elif nombre_modelo == 'LightGBM':
        param_grid = {
            'regressor__num_leaves': [31, 50],
            'regressor__learning_rate': [0.1, 0.05],
            'regressor__n_estimators': [100, 200],
            'regressor__max_depth': [10, 20]
        }
        grid_search = GridSearchCV(
            estimator=pipeline,
            param_grid=param_grid,
            cv=3,
            scoring='neg_mean_absolute_error',
            n_jobs=-1
        )
        grid_search.fit(X_train, y_train)
        best_model = grid_search.best_estimator_
        training_time = time.time() - start_time
        best_params = grid_search.best_params_
    else:
        pipeline.fit(X_train, y_train)
        best_model = pipeline
        training_time = time.time() - start_time
        best_params = None

    mostrar_mensaje(f" {nombre_modelo} entrenado en {training_time:.2f} segundos.", "success")

    # Tiempo de prediccin
    start_pred_time = time.time()
    y_pred = best_model.predict(X_test)
    pred_time = time.time() - start_pred_time

    #  4.5 Evaluacin del Modelo
    rmse = np.sqrt(mean_squared_error(y_test, y_pred))
    mae = mean_absolute_error(y_test, y_pred)
    r2 = r2_score(y_test, y_pred)

    resultados.append({
        'Modelo': nombre_modelo,
        'RMSE': rmse,
        'MAE': mae,
        'R': r2,
        'Tiempo Entrenamiento (s)': training_time,
        'Tiempo Prediccin (s)': pred_time,
        'Mejores Parmetros': best_params
    })

    #  4.6 Guardado de Modelos Entrenados
    joblib.dump(best_model, f'modelo_{nombre_modelo.lower().replace(" ", "_")}.pkl')
    mostrar_mensaje(f" Modelo {nombre_modelo} guardado.", "success")

def analizar_y_validar_modelos():
    start_time = time.time()
    mostrar_mensaje(" Iniciando el anlisis y validacin de modelos...", "info")
    
    try:
        # Lista de modelos y sus archivos
        modelos = {
            'Regresin Lineal': 'modelo_regresin_lineal.pkl',
            'Bosque Aleatorio': 'modelo_bosque_aleatorio.pkl',
            'LightGBM': 'modelo_lightgbm.pkl'
        }
        
        #  5.6 Carga de Datos Preprocesados
        nombre_archivo_datos = 'sprint13_taxi.csv'
        try:
            df = pd.read_csv(nombre_archivo_datos, parse_dates=['datetime'], index_col='datetime')
            mostrar_mensaje(f" Datos cargados correctamente desde '{nombre_archivo_datos}'.", "success")
        except FileNotFoundError:
            mostrar_mensaje(f" Error: El archivo '{nombre_archivo_datos}' no fue encontrado.", "error")
            return
        except Exception as e:
            mostrar_mensaje(f" Error al cargar los datos: {str(e)}", "error")
            return
        
        X = df.drop('num_orders', axis=1)
        y = df['num_orders']
        
        #  5.2 Divisin del Conjunto de Datos
        X_train, X_test, y_train, y_test = train_test_split(
            X, y, test_size=0.1, shuffle=False  # Mantener el orden temporal
        )
        
        # Diccionario para almacenar mtricas
        metricas = {}
        
        # Iterar sobre cada modelo
        for nombre, archivo in modelos.items():
            mostrar_mensaje(f" Evaluando modelo: {nombre}", "info")
            
            #  5.1 Carga del Modelo Entrenado
            try:
                modelo = joblib.load(archivo)
                mostrar_mensaje(f" Modelo '{nombre}' cargado correctamente.", "success")
            except FileNotFoundError:
                mostrar_mensaje(f" Error: El archivo '{archivo}' no fue encontrado.", "error")
                continue
            except Exception as e:
                mostrar_mensaje(f" Error al cargar el modelo '{nombre}': {str(e)}", "error")
                continue
            
            #  5.2 Evaluacin en el Conjunto de Prueba
            y_pred = modelo.predict(X_test)
            rmse = np.sqrt(mean_squared_error(y_test, y_pred))
            mae = mean_absolute_error(y_test, y_pred)
            r2 = r2_score(y_test, y_pred)
            
            # Guardar mtricas
            metricas[nombre] = {
                'RMSE': rmse,
                'MAE': mae,
                'R': r2
            }
            
            mostrar_mensaje(
                f" **{nombre}** - RMSE: {rmse:.4f}, MAE: {mae:.4f}, R: {r2:.4f}",
                "success"
            )
            
            #  5.3 Anlisis de Residuos
            residuals = y_test - y_pred
            
            # Histograma de los residuos
            plt.figure(figsize=(10, 6))
            sns.histplot(residuals, bins=50, kde=True, color='purple')
            plt.title(f'Distribucin de Residuos - {nombre}')
            plt.xlabel('Residuos')
            plt.ylabel('Frecuencia')
            plt.grid(True)
            plt.show()
            
            # Grfico de Residuos vs Predicciones
            plt.figure(figsize=(10, 6))
            sns.scatterplot(x=y_pred, y=residuals, alpha=0.5)
            plt.title(f'Residuos vs Predicciones - {nombre}')
            plt.xlabel('Predicciones')
            plt.ylabel('Residuos')
            plt.axhline(0, color='red', linestyle='--')
            plt.grid(True)
            plt.show()
            
            #  5.5 Importancia de las Caractersticas (Solo para modelos basados en rboles)
            if nombre in ['Bosque Aleatorio', 'LightGBM']:
                try:
                    importances = modelo.named_steps['regressor'].feature_importances_
                    feature_names = modelo.named_steps['preprocessor'].transformers_[0][2]  # Obtener nombres numricos
                    df_importances = pd.DataFrame({
                        'Caracterstica': feature_names,
                        'Importancia': importances
                    }).sort_values(by='Importancia', ascending=False).head(10)
                    
                    plt.figure(figsize=(10, 6))
                    sns.barplot(x='Importancia', y='Caracterstica', data=df_importances, palette='viridis')
                    plt.title(f'Importancia de Caractersticas - {nombre}')
                    plt.xlabel('Importancia')
                    plt.ylabel('Caracterstica')
                    plt.grid(True)
                    plt.show()
                except Exception as e:
                    mostrar_mensaje(f" No se pudo obtener la importancia de caractersticas para {nombre}: {str(e)}", "error")
        
        # Crear DataFrame de mtricas
        df_metricas = pd.DataFrame(metricas).T
        df_metricas = df_metricas.sort_values(by='RMSE')
        
        # Mostrar tabla de mtricas
        display(df_metricas.style.highlight_min(subset=['RMSE', 'MAE'], color='lightgreen').highlight_max(subset=['R'], color='lightgreen').set_caption("Comparacin de Mtricas entre Modelos"))
        
        # Comparacin de Modelos - Grficos
        plt.figure(figsize=(12, 6))
        sns.barplot(x=df_metricas.index, y='RMSE', data=df_metricas, palette='viridis')
        plt.title('Comparacin de RMSE entre Modelos')
        plt.ylabel('RMSE')
        plt.xlabel('Modelo')
        plt.show()
        
        plt.figure(figsize=(12, 6))
        sns.barplot(x=df_metricas.index, y='MAE', data=df_metricas, palette='coolwarm')
        plt.title('Comparacin de MAE entre Modelos')
        plt.ylabel('MAE')
        plt.xlabel('Modelo')
        plt.show()
        
        plt.figure(figsize=(12, 6))
        sns.barplot(x=df_metricas.index, y='R', data=df_metricas, palette='Blues_d')
        plt.title('Comparacin de R entre Modelos')
        plt.ylabel('R')
        plt.xlabel('Modelo')
        plt.show()
        
        # Guardar las mtricas en un archivo CSV
        df_metricas.to_csv('metricas_modelos.csv')
        mostrar_mensaje(" Mtricas de modelos guardadas como 'metricas_modelos.csv'.", "success")
        
        # Identificar el mejor modelo segn RMSE
        mejor_modelo = df_metricas.index[0]
        mostrar_mensaje(f" El mejor modelo segn RMSE es: {mejor_modelo}", "success")
        
    except Exception as e:
        mostrar_mensaje(f" Error durante el anlisis: {str(e)}", "error")
    
    end_time = time.time()
    mostrar_mensaje(f" Anlisis y validacin completados en {end_time - start_time:.2f} segundos.", "success")
